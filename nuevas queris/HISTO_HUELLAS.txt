CREATE TABLE "DBA"."HISTO_HUELLAS" (
	"HISTO_HUE_ID" INTEGER NOT NULL DEFAULT AUTOINCREMENT UNIQUE,
	"HISTO_HUE_CODIGO" VARCHAR(50) NULL,
	"HISTO_HUE_IDENTIDAD" VARCHAR(50) NULL,
	"HISTO_TIPO_PER" VARCHAR(50) NULL,
	"HISTO_HUE_FECHA_AGREGO" "datetime" NULL,
	"HISTO_HUE_FECHA_ELIMINA" "datetime" NULL,
	"HISTO_HUE_MOTIVO_ELIMINA" VARCHAR(500) NULL,
	"HISTO_HUE_USR_ELIMINA" VARCHAR(100) NULL,
	"HISTO_HUE_USR_AGREGO" VARCHAR(100) NULL,
	PRIMARY KEY ( "HISTO_HUE_ID" ASC )
) IN "system";
COMMENT ON TABLE "DBA"."HISTO_HUELLAS" IS 'HISTORICO EN CASO DE BORRAR HUELLAS';
COMMENT ON COLUMN "DBA"."HISTO_HUELLAS"."HISTO_HUE_CODIGO" IS 'el codigo de persona al que pertenecia';
COMMENT ON COLUMN "DBA"."HISTO_HUELLAS"."HISTO_HUE_IDENTIDAD" IS 'identidad a la que pertenecia';
COMMENT ON COLUMN "DBA"."HISTO_HUELLAS"."HISTO_TIPO_PER" IS 'tipo de la persona al que pertenecia la huella';
COMMENT ON COLUMN "DBA"."HISTO_HUELLAS"."HISTO_HUE_MOTIVO_ELIMINA" IS 'motivo del porque elimina la huella';
